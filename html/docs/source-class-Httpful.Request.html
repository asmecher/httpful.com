472:     /**
473:      * Magic method allows for neatly setting other headers in a
474:      * similar syntax as the other setters.  This method also allows
475:      * for the sends* syntax.
476:      * @return Request this
477:      * @param string $method "missing" method name called
478:      *    the method name called should be the name of the header that you
479:      *    are trying to set in camel case without dashes e.g. to set a
480:      *    header for Content-Type you would use contentType() or more commonly
481:      *    to add a custom header like X-My-Header, you would use xMyHeader().
482:      *    To promote readability, you can optionally prefix these methods with
483:      *    "with"  (e.g. withXMyHeader("blah") instead of xMyHeader("blah")).
484:      * @param array $args in this case, there should only ever be 1 argument provided
485:      *    and that argument should be a string value of the header we're setting
486:      */
487:     public function __call($method, $args)
488:     {
489:         // This method supports the sends* methods
490:         // like sendsJSON, sendsForm
491:         //!method_exists($this, $method) &&
492:         if (substr($method, 0, 5) === 'sends') {
493:             $mime = strtolower(substr($method, 5));
494:             if (Mime::supportsMimeType($mime)) {
495:                 $this->sends(Mime::getFullMime($mime));
496:                 return $this;
497:             }
498:             // else {
499:             //     throw new \Exception("Unsupported Content-Type $mime");
500:             // }
501:         }
502:         if (substr($method, 0, 7) === 'expects') {
503:             $mime = strtolower(substr($method, 7));
504:             if (Mime::supportsMimeType($mime)) {
505:                 $this->expects(Mime::getFullMime($mime));
506:                 return $this;
507:             }
508:             // else {
509:             //     throw new \Exception("Unsupported Content-Type $mime");
510:             // }
511:         }
512: 
513:         // This method also adds the custom header support as described in the
514:         // method comments
515:         if (count($args) === 0)
516:             return;
517: 
518:         // Strip the sugar.  If it leads with "with", strip.
519:         // This is okay because: No defined HTTP headers begin with with,
520:         // and if you are defining a custom header, the standard is to prefix it
521:         // with an "X-", so that should take care of any collisions.
522:         if (substr($method, 0, 4) === 'with')
523:             $method = substr($method, 4);
524: 
525:         // Precede upper case letters with dashes, uppercase the first letter of method
526:         $header = ucwords(preg_replace('/[A-Z]/', '-$1', $method));
527:         $this->addHeader($header, $args[0]);

601:     /**
602:      * Does the heavy lifting.  Uses de facto HTTP
603:      * library cURL to set up the HTTP request.
604:      * Note: It does NOT actually send the request
605:      * @return Request $this;
606:      */
607:     private function _curlPrep()
608:     {
609:         // Check for required stuff
610:         if (!isset($this->uri))
611:             throw new \Exception('Attempting to send a request before defining a URI endpoint.');
612: 
613:         $ch = curl_init($this->uri);
614: 
615:         curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $this->method);
616: 
617:         if ($this->hasBasicAuth()) {
618:             curl_setopt($ch, CURLOPT_USERPWD, $this->username . ':' . $this->password);
619:         }
620: 
621:         if ($this->hasClientSideCert()) {
622: 
623:             if (!file_exists($this->client_key))
624:                 throw new \Exception('Could not read Client Key');
625: 
626:             if (!file_exists($this->client_cert))
627:                 throw new \Exception('Could not read Client Certificate');
628: 
629:             curl_setopt($ch, CURLOPT_SSLCERTTYPE,   $this->client_encoding);
630:             curl_setopt($ch, CURLOPT_SSLKEYTYPE,    $this->client_encoding);
631:             curl_setopt($ch, CURLOPT_SSLCERT,       $this->client_cert);
632:             curl_setopt($ch, CURLOPT_SSLKEY,        $this->client_key);
633:             curl_setopt($ch, CURLOPT_SSLKEYPASSWD,  $this->client_passphrase);
634:             // curl_setopt($ch, CURLOPT_SSLCERTPASSWD,  $this->client_cert_passphrase);
635:         }
636: 
637:         curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, $this->strict_ssl);
638: 
639:         curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
640: 
641:         $headers = array("Content-Type: {$this->content_type}");
642: 
643:         $headers[] = !empty($this->expected_type) ?
644:             "Accept: {$this->expected_type}, text/plain" :
645:             "Accept: */*";
646: 
647:         foreach ($this->headers as $header => $value) {
648:             $headers[] = "$header: $value";
649:         }
650:         curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
651: 
652:         if (isset($this->payload)) {
653:             $this->serialized_payload = $this->_serializePayload($this->payload);
654:             curl_setopt($ch, CURLOPT_POSTFIELDS, $this->serialized_payload);
655:         }
656: 
657:         if ($this->_debug) {
658:             curl_setopt($ch, CURLOPT_VERBOSE, true);
659:         }
660: 
661:         curl_setopt($ch, CURLOPT_HEADER, 1);
662: 
663:         // If there are some additional curl opts that the user wants
664:         // to set, we can tack them in here
665:         foreach ($this->additional_curl_opts as $curlopt => $curlval) {
666:             curl_setopt($ch, $curlopt, $curlval);
667:         }
668: 
669:         $this->_ch = $ch;
670: 
671:         return $this;
672:     }
